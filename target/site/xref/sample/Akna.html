11      private ArrayList<Integer>list = new ArrayList<Integer>();
12      private double NumberOfBombs=2.0;
13  
14      /**
15       * Places the number of bombs that given by @NumberOfBombs, than counts the bombs in the neighbours.
16       */
17  
18      public void aknák() {
19          for (int i = 0; i < counter[0].length; i++) {
20              for (int j = 0; j < counter.length; j++) {
21                  list.add(i * 100 + j);
22              }
23          }
24  
25  
26          for (int i = 0; i < NumberOfBombs; i++) {
27              int bomb = (int) (Math.random() * list.size());
28              counter[list.get(bomb) / 100][list.get(bomb) % 100] = BOMB;
29              list.remove(bomb);
30  
31  
32          }
33          for (int i = 0; i < counter[0].length; i++) {
34              for (int j = 0; j < counter.length; j++) {
35                  if (counter[i][j] != BOMB) {
36                      int szomszed = 0;
37                      if (i > 0 && j > 0 && counter[i - 1][j - 1] == BOMB)                                    //bal felső
38                          szomszed++;
39                      if (j > 0 && counter[i][j - 1] == BOMB)                                             //felső
40                          szomszed++;
41                      if (i > 0 && counter[i - 1][j] == BOMB)                                             //bal
42                          szomszed++;
43                      if (j < counter.length - 1 && counter[i][j + 1] == BOMB)                              //alsó
44                          szomszed++;
45                      if (i < counter[0].length - 1 && counter[i + 1][j] == BOMB)                           //jobb
46                          szomszed++;
47                      if (i < counter[0].length - 1 && j < counter.length - 1 && counter[i + 1][j + 1] == BOMB)   //jobb felső
48                          szomszed++;
49                      if (i > 0 && j < counter.length - 1 && counter[i - 1][j + 1] == BOMB)                     //bal alsó
50                          szomszed++;
51                      if (i < counter[0].length - 1 && j > 0 && counter[i + 1][j - 1] == BOMB)                  //jobb alsó
52                          szomszed++;
53  
54                      counter[i][j] = szomszed;
55                  }
56              }
57          }
58      }
59  
60      /**
61       *
62       * @return gives the number of bombs.
63       */
64      public double getNumberOfBombs() {
65          return NumberOfBombs;
