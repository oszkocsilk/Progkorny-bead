11  /**
12   * Blows the bomb is you choose it.
13   */
14  
15  public class BlowedUp implements mezo{
16  
17      private Aknakereso aknakereso;
18  
19      /**
20       *Calls the created @aknakereso.
21       * @param aknakereso the created Aknakereso
22       */
23  
24      public BlowedUp(Aknakereso aknakereso) {
25          this.aknakereso = aknakereso;
26      }
27  
28      /**
29       * This watches when you chose a bomb and shows that is a bomb.
30       */
31      public void Blowed() {
32          for (int x = 0; x < aknakereso.getMezo().length; x++) {
33              for (int y = 0; y < aknakereso.getMezo().length; y++) {
34                  aknakereso.getMezo()[x][y].setDisable(true);
35  
36                  if (counter[x][y] == 666) {
37  
38                      aknakereso.getMezo()[x][y].setText("X");
39                      aknakereso.getMezo()[x][y].setStyle("-fx-background-color: #ff0000");
40                  }
41              }
42          }
43      }
44  
45      /**
46       *
47       * @param title az új ablak címe
48       * @param message az új ablak üzenete
49       */
50  
51      public static void display(String title, String message) {
52          Stage window = new Stage();
53  
54          window.initModality(Modality.APPLICATION_MODAL);
55          window.setTitle(title);
56          window.setMinWidth(250);
57  
58          Label label = new Label();
59          label.setText(message);
60          Button closeButton = new Button("Close this window");
61          closeButton.setOnAction(e -> window.close());
62  
63          VBox layout = new VBox(20);
64          layout.getChildren().addAll(label, closeButton);
65          layout.setAlignment(Pos.CENTER);
66  
67          Scene scene = new Scene(layout);
68          window.setScene(scene);
69          window.show();
70      }
71  }
